rgb = img.convert("RGB") px = rgb.load() w, h = rgb.size alpha = Image.new("L", (w, h), 0) apx = alpha.load() for y in range(h): for x in range(w): r, g, b = px[x, y] # white bg -> alpha 0; blue lines -> alpha high darkness = 255 - min(r, g, b) # amplify line opacity a = max(0, min(255, int((darkness - 12) * 2.2))) apx[x, y] = a # keep original blue lines, but alpha white out out = Image.new("RGBA", (w, h), (0,0,0,0)) out.putdata([(10, 45, 155, apx[x % w, x // w]) for x in range(w*h)]) # crop to content bbox = alpha.point(lambda p: 255 if p > 18 else 0).getbbox() if bbox: left, top, right, bottom = bbox left = max(0, left - pad) top = max(0, top - pad) right = min(w, right + pad) bottom = min(h, bottom + pad) out = out.crop((left, top, right, bottom)) # resize ow, oh = out.size scale = min(max_dim / max(ow, oh), 1) if scale < 1: out = out.resize((int(ow*scale), int(oh*scale)), Image.LANCZOS) bio = io.BytesIO() out.save(bio, format="PNG", optimize=True) return
